CREATE DATABASE XiaoShouTuihuo;

-- 销售退货主表
DROP TABLE IF EXISTS XiaoShouTuihuo;
CREATE TABLE XiaoShouTuihuo(
User_kehu VARCHAR(50),-- 客户
Date_Riqi VARCHAR(50),-- 单据日期 
Address VARCHAR(50),-- 送货地址 
Phone VARCHAR(50),-- 单据号码
Outbound VARCHAR(50),-- 销售出库类型
Currency VARCHAR(50),-- 币别
YNtax VARCHAR(50),-- 单据是否含税
Exvhange VARCHAR(50),-- 汇率
Warehouse VARCHAR(50),-- 仓库
Trading VARCHAR(50),-- 国外贸易
CertificateID VARCHAR(50) primary key, -- 凭证编号 主键
zhidong VARCHAR(50),-- 复核后自动生成发票
a VARCHAR(50),-- 预留
b VARCHAR(50)-- 预留
)

-- 销售退货子表
DROP TABLE IF EXISTS TuiHuo_details;
CREATE TABLE TuiHuo_details(
Columno INT PRIMARY key,-- 栏号 主键
MaterialsID VARCHAR(50) ,-- 物料编号 主键
MaterialsName VARCHAR(50),-- 物料名称
Specification VARCHAR(50),-- 规格型号
UnitName VARCHAR(50),-- 单位名称
Number INT,-- 数量
Discount DECIMAL, -- 折扣前单价
Foldnum DECIMAL,-- 折数
Price INT,-- 单价
Amount INT,-- 金额
Taxrate DECIMAL,-- 税率
Taxe DECIMAL, -- 税额
TaxeAmount DECIMAL,-- 含税金额
LotNumber VARCHAR(50),-- 批号
MaterialsComB VARCHAR(50),-- 物料组合
Gift VARCHAR(50),-- 赠品
Invoice VARCHAR(50),-- 发票明细
NotInvoice VARCHAR(50),-- 未开发票
Remarks VARCHAR(50),-- 分录备注
Single VARCHAR(50),-- 来源单别
SingleNumber VARCHAR(50),-- 来源单号
UserOrder VARCHAR(50),-- 客户订单
BusinessPeople VARCHAR(50),-- 业务人员
BillingStaff VARCHAR(50),-- 制单人员
Department VARCHAR(50),-- 所属部门
Reviewer VARCHAR(50),-- 复核人员
ViewProject VARCHAR(50)-- 所属项目
)
