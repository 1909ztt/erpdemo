-- 销售报价单(主表)
drop table if exists SalesModoule;
CREATE table SalesModoule(
DocumentNumber NVARCHAR(50) PRIMARY key,-- 单据号码关联外键栏号
FormalCustomer NVARCHAR(50),-- 正式客户
DocumentDate NVARCHAR(50),-- 单据日期
DeliveryAddress NVARCHAR(50), -- 送货地址 
Effective NVARCHAR(50), -- 有效日期
Currency NVARCHAR(50), -- 币别
WhetherTheTax NVARCHAR(50), -- 是否含税
Exchange NVARCHAR(50), -- 汇率
Remark NVARCHAR(50), -- 备注
BusinessPersonnel NVARCHAR(50), -- 业务人员
MakingPersonnel NVARCHAR(50), -- 制单人员
Department NVARCHAR(50), -- 所属部门
ReviewingOfficer NVARCHAR(50), -- 复核人员
A NVARCHAR(50), -- 备用1
B NVARCHAR(50) -- 备用2
);

UnitName NVARCHAR(50), -- 单位名称
Amount NVARCHAR(50), -- 数量

FoldTheNumber NVARCHAR(50), -- 折数
UnitPrice NVARCHAR(50), -- 单价
`Sum` NVARCHAR(50), -- 金额
TaxRate NVARCHAR(50), --  税率
AmountOfTax NVARCHAR(50), -- 税额
TaxIncludedAmount NVARCHAR(50), -- 含税金额
Complimentary NVARCHAR(50), -- 赠品
EntryNote NVARCHAR(50), -- 分录备注
A NVARCHAR(50), -- 备用1
B NVARCHAR(50) -- 备用2
);
